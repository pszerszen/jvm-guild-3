String

repeat()

                    
                        strVar.repeat(2) ➡️ strVar + strVar

isBlank()

                    
                        ✅ - "".isBlank()
                        ✅ - " ".isBlank()
                        ✅ - " \n ".isBlank()
                        ❌ - "some string".isBlank()

lines()

                        
                            "foo\nbar\n\nfoo\nbar".lines() ➡️ Stream.of("foo","bar","","foo","bar")

strip()

                        
                            var s = "\t abc \n";
                            s.strip() == s.trim()

strip()

                        
                            var c = '\u2000';
                            var s = c + "abc" + c;
                            s.strip() != s.trim()

stripLeading()

                        
                            var s = "\t abc \n";
                            s.stripLeading() ➡️ "abc \n"

stripTrailing()

                        
                            var s = "\t abc \n";
                            s.stripTrailing() ➡️ "\t abc"

transform()

                        
                            "hello".transform(input -> input + " world!") ➡️ "hello world!"

indent()

                        
                            var multiline = "This is\na multiline\nstring.";
                            var indented = "  This is\n  a multiline\n  string.";
                            multiline.indent(2) ➡️ indented
                            indented.indent(-2) ➡️ multiline
                            indented.indent(-999) ➡️ multiline

CompactNumberFormat

                    
                        var numberFormat = CompactNumberFormat.getCompactNumberInstance(
                                Locale.ENGLISH, Style.SHORT);
                        numberFormat.format(x);

                              x ➡️ output
                            100 ➡️ "100"
                           1000 ➡️ "1K"
                         100000 ➡️ "100K"
                        1000000 ➡️ "1M"
                    
                    
                    
                        var numberFormat = CompactNumberFormat.getCompactNumberInstance(

                        numberFormat.format(x);

                              x ➡️ output

                           1000 ➡️ "1 tysiąc"
                         100000 ➡️ "100 tysięcy"

switch-case

Old style

                    
                        int value;
                        switch (day) {
                        case MONDAY:
                        case FRIDAY:
                        case SUNDAY:
                            value = 6;
                            break;
                        case TUESDAY:
                            value = 7;
                            break;
                        case THURSDAY:
                        case SATURDAY:
                            value = 8;
                            break;
                        case WEDNESDAY:
                            value = 9;
                            break;
                        default:
                            throw new IllegalStateException("What: " + day);

yield

                    
                    int value = switch (day) {
                        case MONDAY, FRIDAY, SUNDAY:
                            yield 6;
                        case TUESDAY:
                            yield 7;
                        case THURSDAY, SATURDAY:
                            yield 8;
                        case WEDNESDAY:
                            yield 9;
                    };

lambda

                    
                    int value = switch (day) {
                        case MONDAY, FRIDAY, SUNDAY -> 6;
                        case TUESDAY -> 7;
                        case THURSDAY, SATURDAY -> 8;
                        case WEDNESDAY -> 9;
                    };
