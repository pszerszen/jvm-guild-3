repeat()

                    
                        strVar.repeat(2) ➡️ strVar + strVar

isBlank()

                    
                        ✅ - "".isBlank()
                        ✅ - " ".isBlank()
                        ✅ - " \n ".isBlank()
                        ❌ - "some string".isBlank()

lines()

                        
                            "foo\nbar\n\nfoo\nbar".lines() ➡️ Stream.of("foo","bar","","foo","bar")

strip()

                        
                            var s = "\t abc \n";
                            s.strip() == s.trim()

strip()

                        
                            var c = '\u2000';
                            var s = c + "abc" + c;
                            s.strip() != s.trim()

stripLeading()

                        
                            var s = "\t abc \n";
                            s.stripLeading() ➡️ "abc \n"

stripTrailing()

                        
                            var s = "\t abc \n";
                            s.stripTrailing() ➡️ "\t abc"
